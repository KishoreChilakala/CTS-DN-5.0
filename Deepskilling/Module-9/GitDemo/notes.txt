--- Doc 1: Git Setup, First Commit, Push & Pull ---
# Verify Git installation
git --version

# Configure user identity (global)

# Check configuration

# Configure Notepad++ as default editor

# Create and initialise local repo

# Create a file and check status

# Stage and commit with multi-line message (opens editor)

# Verify working dir matches local repo
git status            # should show "nothing to commit"

# Link to remote and push
git remote add origin <your-gitlab-repo-url>
git pull origin master
git push -u origin master

--- Doc 3: Branching & Merging ---
# Create and list branches
git branch            # shows current branch
git checkout -b GitNewBranch   # create & switch

# Add work on branch
echo "branch work" > feature.txt
git add feature.txt
git commit -m "Add feature.txt on GitNewBranch"
git status

# Diff branch vs master (CLI)
git diff master..GitNewBranch

# Merge branch into master
git checkout master
git merge GitNewBranch

# Visualise after merge
git log --oneline --graph --decorate

# Delete merged branch
git branch -d GitNewBranch

--- Doc 4: Conflict Resolution ---
# Setup conflict: same file edited differently on master and branch
git checkout -b GitWork
# (edit hello.xml in branch)
git add hello.xml && git commit -m "GitWork changes"

git checkout master
# (edit hello.xml differently on master)
git add hello.xml && git commit -m "Master changes"

# Merge — will conflict
git merge GitWork     # CONFLICT in hello.xml

# View conflict markers
git diff              # shows <<<<<< HEAD / ======= / >>>>>>> GitWork

# Resolve: edit hello.xml, remove markers, keep correct content
git add hello.xml
git commit -m "Resolve merge conflict in hello.xml"

# Add backup/orig files generated by merge tools to .gitignore
echo "*.orig" >> .gitignore
git add .gitignore
git commit -m "Ignore *.orig merge backup files"

git branch -d GitWork
git log --oneline --graph --decorate

--- Doc 5: Cleanup & Push to Remote ---
# Verify master is clean

# List all branches
git branch -a

# Pull latest from remote
git pull origin master

# Push all pending commits to remote
git push origin master
